187 # Calculates voronoi partition for current flock/pack behaviour
188 def calc_voronoi_partitioning(flock, pack):
189
190 """For a given pair of flock and pack groups, the voronoi partition is calculated.
191
192 Each dog agent has their subflock cleared, then reset with closest sheep agents.
193
194 Each sheep recalculates their nearest dog """
195 for dog in pack:
196 dog.empty_sub_flock()
197
198 for sheep in flock:
199 min_dist = 10000
200 for dog in pack:
201 dist = np.linalg.norm(sheep.position - dog.position)
202 if dist < min_dist:
203 min_dist = dist
204 sheep.set_closest_dog(dog)
205 sheep.closest_dog.add_sheep_to_sub_flock(sheep)
206 # end function

307 # Function repeatedly called to decide if an agent should move to a destination, or if it already has reached destination.
308 # Calls MoveToPoint on agent if displacement from target is greater than threshold, otherwise HaltAgent is called.
309 # HaltAgent sends a command to physical robot to stop moving
310 def MoveToPointDecision(agent: Agent, movePos, cfg):
311 """ Function repeatedly called to decide if an agent should move to a destination, or if it already has reached destination.
312 Calls function MoveToPoint on specified agent if its displacement from a target is greater than a threshold, otherwise the function HaltAgent is called.
313
314 Function HaltAgent sends a command to physical robot to stop moving
315
316 Parameters:
317 agent: Subject agent of type agent
318 movePos: Target screenspace coordinates
319 cfg: Config file dictionary
320 """
321 point_x = movePos[0]
322 point_y = movePos[1]
323 agentPos = np.array([agent.position[0], agent.position[1]])
324 # TODO - Set distance to a variable in the cfg file instead of a magic number
325 if(np.linalg.norm(movePos - agentPos) > 85):
326 agent.MoveToPoint(point_x=point_x, point_y=point_y,
327 screen=screen, agents=agents, cfg=cfg)
328 else:
329 agent.HaltAgent(screen=screen)
330
331 # Extended version of MoveToPointDecision. Takes a list of destinations and moves agent towards each in order. Stops and returns 1 when finished path.
332 # Path is usually defined by points calculated with the pathfinding feature.
333 def FollowPathDecision(agent: Agent, path, cfg):
334
335 """Extended version of MoveToPointDecision. Takes a list of destinations and moves agent towards each destination in order. Stops and returns 1 when agent finishes path.
336
337 Parameters:
338 agent: Subject agent of type agent
339 path: List of screenspace coordinate pairs
340 cfg: Config file dictionary
341 """
342
343 pathFindingTileRadius = cfg['world_width'] / cfg['path_finding_width']
344
345 if(len(path) == 0):
346 agent.HaltAgent(screen=screen)
347 return 1
348
349 if(len(path) > 0):
350 agentPos = np.array([agent.position[0], agent.position[1]])
351 targetPos = path[0]
352
353 if(len(path) == 1):
354 if(not cfg['event_driven']):
355 pathFindingTileRadius = 8
356
357 if(np.linalg.norm(targetPos - agentPos) > pathFindingTileRadius):
358 agent.MoveToPoint(
359 point_x=targetPos[0], point_y=targetPos[1], screen=screen, agents=agents, cfg=cfg)
360 else:
361 path = path[1:]
362 agent.SetPath(path)
363
364 else:
365 agent.HaltAgent(screen=screen)
366 return 1
367
368 return 0

475 def isInRectangle(centerX, centerY, radius, x, y):
476
477 return x >= centerX - radius and x <= centerX + radius and y >= centerY - radius and y <= centerY + radius
478
479 # test if coordinate (x, y) is within a radius from coordinate (center_x, center_y)
480
481 # Function used to determine if a user's mouse click is inside one of the agents. Used in the process of user manually removing an agent
482 def isPointInCircle(centerX, centerY, radius, x, y):
483 if(isInRectangle(centerX, centerY, radius, x, y)):
484 dx = centerX - x
485 dy = centerY - y
486 dx *= dx
487 dy *= dy
488 distanceSquared = dx + dy
489 radiusSquared = radius * radius
490 return distanceSquared <= radiusSquared
491 return False
